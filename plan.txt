## A100 Commands (run in order)

------------------------------------------------------------
### PHASE 1: Reorganize seokmin directory
------------------------------------------------------------
# Before replacing robomme_policy_learning, move docker tars out first
mkdir -p ~/seokmin/docker_images
mv ~/seokmin/robomme_policy_learning/mme_vla_cuda12.8.tar.gz ~/seokmin/docker_images/
mv ~/seokmin/robomme_policy_learning/nvidia_cuda12.8_base.tar.gz ~/seokmin/docker_images/

# Now replace robomme_policy_learning with the complete recursive clone (from USB)
# (USB has the full repo with third_party/robomme_benchmark populated)
rm -rf ~/seokmin/robomme_policy_learning
cp -r /mnt/usb/seokmin/robomme_policy_learning_full ~/seokmin/robomme_policy_learning
# → third_party/robomme_benchmark/ should now be populated

------------------------------------------------------------
### PHASE 2: Set up host directory structure
------------------------------------------------------------
# Set up OPENPI_DATA_HOME
mkdir -p ~/openpi_data
cp -r ~/seokmin/openpi_cache/openpi-assets ~/openpi_data/
cp -r ~/seokmin/openpi_cache/big_vision ~/openpi_data/
cp -r ~/seokmin/pi05_vision_encoder ~/openpi_data/pi05_vision_encoder
chmod -R 755 ~/openpi_data

# Set up Qwen HF cache (TRANSFORMERS_OFFLINE needs this exact structure)
mkdir -p ~/hf_cache/hub
cp -r ~/seokmin/Qwen3-VL-4B-Instruct \
    ~/hf_cache/hub/models--Qwen--Qwen3-VL-4B-Instruct

# Copy raw dataset into repo data dir
mkdir -p ~/seokmin/robomme_policy_learning/data
cp -r ~/seokmin/robomme_data_h5 \
    ~/seokmin/robomme_policy_learning/data/robomme_data_h5

# Create runs dir (checkpoints + eval results go here, persisted outside Docker)
mkdir -p ~/seokmin/robomme_policy_learning/runs/ckpts
mkdir -p ~/seokmin/robomme_policy_learning/runs/evaluation
mkdir -p ~/seokmin/robomme_policy_learning/runs/assets/mme_vla_suite

# Add to ~/.bashrc
echo 'export OPENPI_DATA_HOME=~/openpi_data' >> ~/.bashrc
source ~/.bashrc

------------------------------------------------------------
### PHASE 3: Load Docker & verify GPU
------------------------------------------------------------
docker load < ~/seokmin/docker_images/mme_vla_cuda12.8.tar.gz
docker load < ~/seokmin/docker_images/nvidia_cuda12.8_base.tar.gz

# Verify GPU
docker run --rm --gpus all nvidia/cuda:12.8.0-base-ubuntu24.04 nvidia-smi

------------------------------------------------------------
### PHASE 4: Start Docker container (keep alive for all steps)
------------------------------------------------------------
# Run from ~/seokmin/robomme_policy_learning
cd ~/seokmin/robomme_policy_learning

docker run --rm -it --gpus all \
  -e NVIDIA_DRIVER_CAPABILITIES=compute,graphics,utility,video \
  -e OPENPI_DATA_HOME=/openpi_data \
  -e HF_HOME=/hf_cache \
  -e TRANSFORMERS_OFFLINE=1 \
  -e WANDB_MODE=offline \
  -v ~/openpi_data:/openpi_data \
  -v ~/hf_cache:/hf_cache \
  -v ~/seokmin/robomme_policy_learning/data:/app/data \
  -v ~/seokmin/robomme_policy_learning/runs:/app/runs \
  mme_vla:cuda12.8

# All commands below are run INSIDE this container

------------------------------------------------------------
### PHASE 5: Preprocess data (inside Docker)
------------------------------------------------------------
# 5-1. Decompress raw h5 (~2-3 hrs)
uv run scripts/tarxz_h5.py decompress \
    --input_dir data/robomme_data_h5 \
    --jobs 16 \
    --remove_archive

# 5-2. Build preprocessed pkl dataset (~2-3 hrs)
uv run scripts/build_dataset.py \
    --dataset_type robomme_pkl \
    --raw_data_path data/robomme_data_h5 \
    --preprocessed_data_path data/robomme_preprocessed_data

# 5-3. Copy reference norm_stats and compute (~3 min)
cp assets/norm_stats.json runs/assets/mme_vla_suite/
uv run scripts/compute_norm_stats.py \
    --config-name mme_vla_suite \
    --repo-id robomme \
    --dataset-path data/robomme_preprocessed_data

------------------------------------------------------------
### PHASE 6: Train FrameSamp+Perceptual (perceptual-framesamp-modul)
------------------------------------------------------------
# Inside Docker
CUDA_VISIBLE_DEVICES=0,1,2,3 XLA_PYTHON_CLIENT_MEM_FRACTION=0.95 \
uv run scripts/train.py mme_vla_suite \
    --exp-name=perceptual-framesamp-modul \
    --batch-size=64 \
    --num-workers=4 \
    --fsdp-devices=4 \
    --dataset-path=data/robomme_preprocessed_data \
    --model.use_history \
    --model.history_config="perceptual-framesamp-modul.yaml"

# Checkpoint → runs/ckpts/mme_vla_suite/perceptual-framesamp-modul/

------------------------------------------------------------
### PHASE 7: Evaluate FrameSamp+Perceptual
