# 4090 Evaluation Setup
# 목적: A100 훈련 중에 회사 4090에서 공개 체크포인트로 evaluation 미리 진행
#
# 환경:
#   - 회사 4090 (2x RTX 4090): Docker + NVIDIA driver 있음, 이미지 없음
#   - A100: 훈련 중, 같은 네트워크. 4090에서 접속: ssh dsa@192.168.0.5
#   - 외부망 컴퓨터: HuggingFace에서 체크포인트 직접 다운로드 → 단독망 이동 (직접 처리)
#
# 4090 기본 작업 디렉토리: ~/seokmin/memory/
#
# 디렉토리 구조:
#   ~/seokmin/memory/
#   ├── openpi_data/                              ← A100에서 scp
#   ├── hf_cache/                                 ← A100에서 scp (Phase 10만 필요)
#   ├── third_party/
#   │   └── robomme_benchmark/                    ← A100에서 scp → 컨테이너에 마운트
#   └── runs/
#       ├── ckpts/
#       │   └── mme_vla_suite/
#       │       └── perceptual-framesamp-modul/
#       │           ├── history_config.txt         ← ⚠️ 필수! 외부망에서 다운
#       │           └── 79999/                    ← 외부망 컴퓨터에서 다운받은 체크포인트
#       └── assets/
#           └── mme_vla_suite/
#               └── norm_stats.json               ← A100에서 scp

=============================================================
## Docker Quick Start (컨테이너 재시작할 때마다)
=============================================================
docker run --rm -it --gpus all \
  --ipc=host \
  --security-opt seccomp=unconfined \
  -e NVIDIA_DRIVER_CAPABILITIES=compute,graphics,utility,video \
  -e OPENPI_DATA_HOME=/openpi_data \
  -e HF_HOME=/hf_cache \
  -e TRANSFORMERS_OFFLINE=1 \
  -e WANDB_MODE=offline \
  -v ~/seokmin/memory/openpi_data:/openpi_data \
  -v ~/seokmin/memory/hf_cache:/hf_cache \
  -v ~/seokmin/memory/runs:/app/runs \
  -v ~/seokmin/memory/scripts:/app/scripts \
  -v ~/seokmin/memory/third_party/robomme_benchmark:/app/third_party/robomme_benchmark \
  mme_vla:cuda12.8 bash

# 진입 후 항상 실행
